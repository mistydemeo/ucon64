USAGE: flc [OPTION[MODIFIERS]] ARG1 ARG2 ARG3

  -c		create a new file-listing;
  		$ARG1=DIRECTORY $ARG2=NEW_FILELISTING
  -c2		create a new file-listing but take the FILE_ID.DIZ out of an
  		existing file-listing;
		$ARG1=DIRECTORY $ARG2=EXISTING_FILELISTING $ARG3=NEW_FILELISTING
  -s		resort an existing file-listing;
		$ARG1=FILELISTING $ARG2=NEW_FILELISTING
TODO:  -n	nuke (be careful); $ARG1=FILENAME
TODO:  -rn	rename all files in DIRECTORY to lowercase; $ARG1=DIRECTORY
$0=z,
  -z		zippy-search file-listing; $ARG1=PATTERN $ARG2=FILELISTING
modifiers:
  -t		also check/test every archive/file in DIRECTORY
		flags: N=not checked (default), P=passed, F=failed
  -h		output as HTML document with links to the files
TODO:  -ed	use editor for archives/files w/o FILE_ID.DIZ
  -ap		append at NEW_FILELISTING
  -d		sort chronological
  -a		sort alphabetical
  -b		sort by byte size
  -fr		sort reverse
  -k		show sizes in kilobytes

intern FILE_ID.DIZ extract routines:
$0=txtextract,
  -x		extract FILE_ID.DIZ from plain text; $ARG1=PLAIN_TXT_FILE
