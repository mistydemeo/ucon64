Usage: flc [OPTION]... [FILE]...

   -t           sort by modification time
   -X           sort alphabetical
   -S           sort by byte size
  --fr          sort reverse
   -k           show sizes in kilobytes
  --bbs         output as BBS style filelisting (default)
  --html        output as HTML document with links to the files
   -o=FILE      write output into FILE
   -c           also test every possible archive for errors
                  return flags: N=not checked (default), P=passed, F=failed
  --cache=CACHE get a default file_id.diz from CACHE filelisting
                  (searches for filename)
   -R           scan subdirectories recursively
  --version     output version information and exit
  --help        display this help and exit
